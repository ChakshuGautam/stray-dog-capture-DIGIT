# DA-05: View Fraud Indicators

[‚Üê Back to User Stories Master](../../design-outputs/02-user-stories.md) | [‚Üê Back to Admin Stories](./index.md)

## User Story

**As a** District Administrator,
**I want to** see fraud indicators (rejection rate by teacher, GPS spoofing incidents),
**So that** I can identify suspicious behavior and take appropriate action.

## Description

Fraud monitoring is essential for program integrity. This dashboard provides visibility into potential abuse patterns:
- Teachers with high rejection rates
- GPS spoofing attempts
- Unusual submission patterns
- Duplicate submission clusters

## Acceptance Criteria

### Functional Requirements
- [ ] Dashboard showing fraud risk score by teacher
- [ ] GPS spoofing alerts with evidence
- [ ] Unusual activity patterns (time, location, frequency)
- [ ] Rejection rate trends by teacher
- [ ] Drill-down into specific incidents
- [ ] Ability to flag teachers for review
- [ ] Historical fraud incident log

### Non-Functional Requirements
- [ ] Near real-time alerts (< 15 min delay)
- [ ] Audit trail for all admin actions
- [ ] Privacy-compliant data handling

## Technical Implementation

### Fraud Detection Service
```javascript
// src/services/admin/FraudDetectionService.js

class FraudDetectionService {
  static RISK_THRESHOLDS = {
    HIGH: 80,      // Red
    MEDIUM: 60,    // Orange
    LOW: 40        // Yellow
  };

  static FRAUD_INDICATORS = {
    GPS_SPOOFING: {
      code: 'GPS_SPOOFING',
      weight: 30,
      description: 'GPS coordinates show suspicious patterns'
    },
    HIGH_REJECTION: {
      code: 'HIGH_REJECTION',
      weight: 25,
      threshold: 30 // 30% rejection rate
    },
    UNUSUAL_FREQUENCY: {
      code: 'UNUSUAL_FREQUENCY',
      weight: 20,
      threshold: 10 // 10 submissions per hour
    },
    DUPLICATE_RATE: {
      code: 'DUPLICATE_RATE',
      weight: 15,
      threshold: 20 // 20% duplicate rate
    },
    LOCATION_MISMATCH: {
      code: 'LOCATION_MISMATCH',
      weight: 10,
      threshold: 500 // 500m between selfie and dog photo
    }
  };

  static async getActiveAlerts(districtId, severity = null) {
    const params = new URLSearchParams({ districtId });
    if (severity) params.append('severity', severity);

    const response = await fetch(`/api/admin/fraud/alerts?${params}`);
    return response.json();
  }

  static async getHighRiskTeachers(districtId, limit = 20) {
    const response = await fetch(
      `/api/admin/fraud/high-risk-teachers?districtId=${districtId}&limit=${limit}`
    );
    return response.json();
  }

  static async getGPSAnomalies(districtId, startDate, endDate) {
    const params = new URLSearchParams({
      districtId,
      startDate: startDate.toISOString(),
      endDate: endDate.toISOString()
    });

    const response = await fetch(`/api/admin/fraud/gps-anomalies?${params}`);
    return response.json();
  }

  static async getRejectionTrends(districtId, startDate, endDate) {
    const params = new URLSearchParams({
      districtId,
      startDate: startDate.toISOString(),
      endDate: endDate.toISOString()
    });

    const response = await fetch(`/api/admin/fraud/rejection-trends?${params}`);
    return response.json();
  }

  static async flagTeacher(teacherId, reason, adminId) {
    const response = await fetch('/api/admin/fraud/flag-teacher', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ teacherId, reason, flaggedBy: adminId })
    });
    return response.json();
  }

  static calculateRiskScore(teacherStats) {
    let score = 0;

    // GPS spoofing incidents
    if (teacherStats.gpsSpoofingCount > 0) {
      score += this.FRAUD_INDICATORS.GPS_SPOOFING.weight * Math.min(teacherStats.gpsSpoofingCount, 3);
    }

    // High rejection rate
    if (teacherStats.rejectionRate > this.FRAUD_INDICATORS.HIGH_REJECTION.threshold) {
      const excessRate = teacherStats.rejectionRate - this.FRAUD_INDICATORS.HIGH_REJECTION.threshold;
      score += this.FRAUD_INDICATORS.HIGH_REJECTION.weight * (1 + excessRate / 20);
    }

    // Unusual submission frequency
    if (teacherStats.maxHourlySubmissions > this.FRAUD_INDICATORS.UNUSUAL_FREQUENCY.threshold) {
      score += this.FRAUD_INDICATORS.UNUSUAL_FREQUENCY.weight;
    }

    // High duplicate rate
    if (teacherStats.duplicateRate > this.FRAUD_INDICATORS.DUPLICATE_RATE.threshold) {
      score += this.FRAUD_INDICATORS.DUPLICATE_RATE.weight;
    }

    // Location mismatches
    if (teacherStats.locationMismatches > 0) {
      score += this.FRAUD_INDICATORS.LOCATION_MISMATCH.weight * Math.min(teacherStats.locationMismatches, 5);
    }

    return Math.min(Math.round(score), 100);
  }

  static getRiskLevel(score) {
    if (score >= this.RISK_THRESHOLDS.HIGH) return { level: 'high', color: 'üî¥', label: 'High Risk' };
    if (score >= this.RISK_THRESHOLDS.MEDIUM) return { level: 'medium', color: 'üü†', label: 'Medium Risk' };
    if (score >= this.RISK_THRESHOLDS.LOW) return { level: 'low', color: 'üü°', label: 'Low Risk' };
    return { level: 'normal', color: 'üü¢', label: 'Normal' };
  }

  static getAlertSeverity(alert) {
    if (alert.type === 'GPS_SPOOFING') return 'HIGH';
    if (alert.type === 'UNUSUAL_FREQUENCY' && alert.count > 20) return 'HIGH';
    if (alert.type === 'CLUSTER_DUPLICATE') return 'LOW';
    return 'MEDIUM';
  }
}

export default FraudDetectionService;
```
