## User Story

**As a** State Administrator,
**I want to** view Municipal Corporation action rates by district,
**So that** I can evaluate MC effectiveness and identify resource needs.

## Description

Monitoring Municipal Corporation (MC) performance helps state administrators:
- Evaluate capture effectiveness across districts
- Identify under-resourced MC offices
- Track response times and resolution rates
- Plan resource allocation and training
- Measure program impact through MC actions

## Acceptance Criteria

### Functional Requirements
- [ ] Dashboard showing MC performance by district
- [ ] Metrics: capture rate, UTL rate, average response time
- [ ] Comparison of MC offices within a district
- [ ] Time-based analysis (daily, weekly, monthly trends)
- [ ] MC officer workload distribution
- [ ] Resolution type breakdown
- [ ] Pending case backlog by MC
- [ ] Export for MC management reporting

### Non-Functional Requirements
- [ ] Data refreshes every 15 minutes
- [ ] Support for filtering by district, block, time period
- [ ] Drill-down from state to MC level

```javascript
// src/services/stateAdmin/MCActionService.js

class MCActionService {
  static RESOLUTION_TYPES = {
    CAPTURED: { code: 'CAPTURED', label: 'Dog Captured', icon: 'ğŸ•' },
    OWNER_CLAIMED: { code: 'OWNER_CLAIMED', label: 'Owner Claimed', icon: 'ğŸ ' },
    UTL: { code: 'UTL', label: 'Unable to Locate', icon: 'âŒ' }
  };

  static RESPONSE_WINDOWS = {
    WITHIN_24H: { hours: 24, label: 'Within 24 hours', status: 'excellent' },
    ONE_TO_THREE: { hours: 72, label: '1-3 days', status: 'good' },
    THREE_TO_FIVE: { hours: 120, label: '3-5 days', status: 'acceptable' },
    FIVE_TO_SEVEN: { hours: 168, label: '5-7 days', status: 'delayed' },
    OVER_SEVEN: { hours: Infinity, label: '7+ days', status: 'critical' }
  };

  static WORKLOAD_THRESHOLDS = {
    OPTIMAL: 150,
    MODERATE: 175,
    STRAINED: 200,
    CRITICAL: 225
  };

  static async getStatewideMCMetrics() {
    const response = await fetch('/api/state-admin/mc/metrics', {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async getDistrictMCPerformance(sortBy = 'captureRate', order = 'desc') {
    const params = new URLSearchParams({ sortBy, order });
    const response = await fetch(`/api/state-admin/mc/by-district?${params}`, {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async getMCTrends(months = 12) {
    const response = await fetch(`/api/state-admin/mc/trends?months=${months}`, {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async getResolutionBreakdown() {
    const response = await fetch('/api/state-admin/mc/resolution-breakdown', {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async getResponseTimeDistribution() {
    const response = await fetch('/api/state-admin/mc/response-times', {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async getPendingBacklog() {
    const response = await fetch('/api/state-admin/mc/pending-backlog', {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async getWorkloadAnalysis() {
    const response = await fetch('/api/state-admin/mc/workload', {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async getDistrictMCDetails(districtId) {
    const response = await fetch(`/api/state-admin/mc/district/${districtId}`, {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async requestAdditionalResources(districtId, justification) {
    const response = await fetch('/api/state-admin/mc/resource-request', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${getAuthToken()}`
      },
      body: JSON.stringify({ districtId, justification })
    });
    return response.json();
  }

  // Helper methods
  static calculateCaptureRate(captured, verified) {
    if (verified === 0) return 0;
    return Math.round((captured / verified) * 1000) / 10;
  }

  static getWorkloadStatus(casesPerOfficer) {
    if (casesPerOfficer <= this.WORKLOAD_THRESHOLDS.OPTIMAL) return 'optimal';
    if (casesPerOfficer <= this.WORKLOAD_THRESHOLDS.MODERATE) return 'moderate';
    if (casesPerOfficer <= this.WORKLOAD_THRESHOLDS.STRAINED) return 'strained';
    return 'critical';
  }

  static getWorkloadColor(status) {
    const colors = {
      optimal: '#4caf50',
      moderate: '#ffeb3b',
      strained: '#ff9800',
      critical: '#f44336'
    };
    return colors[status] || '#9e9e9e';
  }

  static formatDays(hours) {
    const days = hours / 24;
    if (days < 1) return `${hours} hours`;
    return `${days.toFixed(1)} days`;
  }

  static getResponseWindowStatus(avgHours) {
    if (avgHours <= 24) return 'excellent';
    if (avgHours <= 72) return 'good';
    if (avgHours <= 120) return 'acceptable';
    if (avgHours <= 168) return 'delayed';
    return 'critical';
  }
}

export default MCActionService;
```
