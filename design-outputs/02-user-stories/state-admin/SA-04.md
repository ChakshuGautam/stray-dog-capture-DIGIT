# SA-04: Fraud Analytics

[‚Üê Back to User Stories Master](../../design-outputs/02-user-stories.md) | [‚Üê Back to Admin Stories](./index.md)

## User Story

**As a** State Administrator,
**I want to** view fraud analytics with pattern detection and anomaly alerts,
**So that** I can identify and address systematic fraud across the state.

## Description

State-level fraud analytics aggregates fraud patterns across all districts to:
- Identify cross-district fraud networks
- Detect emerging fraud patterns early
- Measure effectiveness of fraud prevention
- Allocate investigation resources efficiently
- Develop policy responses to new fraud types

## Acceptance Criteria

### Functional Requirements
- [ ] State-level fraud metrics dashboard
- [ ] Cross-district pattern detection
- [ ] Anomaly detection with ML-based alerts
- [ ] Fraud trend analysis over time
- [ ] Heat map of fraud concentration
- [ ] Teacher fraud networks visualization
- [ ] Fraud investigation queue management
- [ ] Export fraud reports for law enforcement

### Non-Functional Requirements
- [ ] Real-time anomaly detection (< 1 hour latency)
- [ ] Pattern detection across 1M+ records
- [ ] Secure access for investigation teams
- [ ] Audit trail for all fraud-related actions

class StateFraudAnalyticsService {
  static FRAUD_TYPES = {
    DUPLICATE_PHOTO: { code: 'DUPLICATE_PHOTO', label: 'Duplicate Photos', icon: 'üñº' },

      headers: { 'Authorization': `Bearer ${getAuthToken()}` }

    return response.json();

    });
    return response.json();

  static async getFraudTrends(months = 12) {
    const response = await fetch(`/api/state-admin/fraud/trends?months=${months}`, {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async getActiveAlerts() {
    const response = await fetch('/api/state-admin/fraud/alerts', {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async getFraudNetworks() {
    const response = await fetch('/api/state-admin/fraud/networks', {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async getInvestigationQueue(page = 1, limit = 20) {
    const params = new URLSearchParams({ page: page.toString(), limit: limit.toString() });
    const response = await fetch(`/api/state-admin/fraud/investigations?${params}`, {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async assignInvestigation(caseId, teamId) {
    const response = await fetch(`/api/state-admin/fraud/investigations/${caseId}/assign`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${getAuthToken()}`
      },
      body: JSON.stringify({ teamId })
    });
    return response.json();
  }

  static async dismissAlert(alertId, reason) {
    const response = await fetch(`/api/state-admin/fraud/alerts/${alertId}/dismiss`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${getAuthToken()}`
      },
      body: JSON.stringify({ reason })
    });
    return response.json();
  }

  static async confirmFraud(caseId, details) {
    const response = await fetch(`/api/state-admin/fraud/investigations/${caseId}/confirm`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${getAuthToken()}`
      },
      body: JSON.stringify(details)
    });
    return response.json();
  }

  static async exportFraudReport(filters) {
    const params = new URLSearchParams(filters);
    const response = await fetch(`/api/state-admin/fraud/export?${params}`, {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.blob();
  }

  // Network analysis helpers
  static buildNetworkGraph(connections) {
    const nodes = new Map();
    const edges = [];

    connections.forEach(conn => {
      if (!nodes.has(conn.teacher1Id)) {
        nodes.set(conn.teacher1Id, {
          id: conn.teacher1Id,
          name: conn.teacher1Name,
          submissions: conn.teacher1Submissions
        });
      }
      if (!nodes.has(conn.teacher2Id)) {
        nodes.set(conn.teacher2Id, {
          id: conn.teacher2Id,
          name: conn.teacher2Name,
          submissions: conn.teacher2Submissions
        });
      }
      edges.push({
        from: conn.teacher1Id,
        to: conn.teacher2Id,
        type: conn.connectionType,
        weight: conn.connectionStrength
      });
    });

    return {
      nodes: Array.from(nodes.values()),
      edges
    };
  }

  static calculateNetworkRisk(network) {
    const { nodes, edges } = network;
    const avgConnections = edges.length / nodes.length;
    const totalSubmissions = nodes.reduce((sum, n) => sum + n.submissions, 0);

    // Risk score based on network density and submissions
    return Math.min(1, (avgConnections * 0.3) + (totalSubmissions / 100 * 0.7));
  }
}

export default StateFraudAnalyticsService;

  static ANOMALY_TYPES = {
    SUBMISSION_SPIKE: 'submission_spike',
    GPS_CLUSTER: 'gps_cluster',
    TIME_PATTERN: 'time_pattern',
    PHOTO_SIMILARITY: 'photo_similarity',
    NETWORK_FORMATION: 'network_formation'
  };

  static DETECTION_CONFIG = {
    submission_spike: {
      threshold: 3.0,  // Standard deviations
      window: 48,      // Hours
      minSamples: 10
    },
    gps_cluster: {
      maxDistance: 10, // Meters
      minPoints: 3,
      timeWindow: 24   // Hours
    },
    time_pattern: {
      intervalThreshold: 60,   // Seconds
      consecutiveCount: 5
    },
    photo_similarity: {
      hashDistance: 5,    // Hamming distance
      timeWindow: 168     // Hours (1 week)
    }
  };

  static async runAnomalyDetection() {
    const response = await fetch('/api/state-admin/fraud/anomaly-detection/run', {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async getDetectionResults(runId) {
    const response = await fetch(`/api/state-admin/fraud/anomaly-detection/${runId}`, {
      headers: { 'Authorization': `Bearer ${getAuthToken()}` }
    });
    return response.json();
  }

  static async configureDetection(anomalyType, config) {
    const response = await fetch('/api/state-admin/fraud/anomaly-detection/configure', {
      method: 'PUT',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${getAuthToken()}`
      },
      body: JSON.stringify({ anomalyType, config })
    });
    return response.json();
  }

  // Client-side anomaly detection helpers
  static detectSubmissionSpike(submissions, config = this.DETECTION_CONFIG.submission_spike) {
    const hourlyGroups = this.groupByHour(submissions, config.window);
    const mean = this.calculateMean(hourlyGroups);
    const stdDev = this.calculateStdDev(hourlyGroups, mean);

    const anomalies = [];
    hourlyGroups.forEach((count, hour) => {
      if (count > mean + (config.threshold * stdDev)) {
        anomalies.push({
          type: 'submission_spike',
          hour,
          count,
          deviation: (count - mean) / stdDev,
          threshold: config.threshold
        });
      }
    });

    return anomalies;
  }

  static detectGPSCluster(submissions, config = this.DETECTION_CONFIG.gps_cluster) {
    const clusters = [];
    const points = submissions.map(s => ({
      id: s.id,
      teacherId: s.teacherId,
      lat: s.gpsCoordinates.latitude,
      lng: s.gpsCoordinates.longitude,
      timestamp: s.timestamp
    }));

    // Simple clustering algorithm (DBSCAN-like)
    const visited = new Set();

    points.forEach(point => {
      if (visited.has(point.id)) return;

      const neighbors = points.filter(p => {
        if (p.id === point.id) return false;
        const distance = this.haversineDistance(point, p);
        const timeDiff = Math.abs(point.timestamp - p.timestamp) / 3600000;
        return distance <= config.maxDistance && timeDiff <= config.timeWindow;
      });

      if (neighbors.length >= config.minPoints - 1) {
        const cluster = [point, ...neighbors];
        cluster.forEach(p => visited.add(p.id));

        clusters.push({
          type: 'gps_cluster',
          center: { lat: point.lat, lng: point.lng },
          points: cluster,
          teacherIds: [...new Set(cluster.map(p => p.teacherId))],
          submissionCount: cluster.length
        });
      }
    });

    return clusters;
  }

  static haversineDistance(point1, point2) {
    const R = 6371e3; // Earth's radius in meters
    const œÜ1 = point1.lat * Math.PI / 180;
    const œÜ2 = point2.lat * Math.PI / 180;
    const ŒîœÜ = (point2.lat - point1.lat) * Math.PI / 180;
    const ŒîŒª = (point2.lng - point1.lng) * Math.PI / 180;

    const a = Math.sin(ŒîœÜ / 2) * Math.sin(ŒîœÜ / 2) +
              Math.cos(œÜ1) * Math.cos(œÜ2) *
              Math.sin(ŒîŒª / 2) * Math.sin(ŒîŒª / 2);
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));

    return R * c;
  }

  static groupByHour(submissions, windowHours) {
    const now = Date.now();
    const groups = new Map();

    for (let i = 0; i < windowHours; i++) {
      groups.set(i, 0);
    }

    submissions.forEach(s => {
      const hourAgo = Math.floor((now - s.timestamp) / 3600000);
      if (hourAgo < windowHours && groups.has(hourAgo)) {
        groups.set(hourAgo, groups.get(hourAgo) + 1);
      }
    });

    return Array.from(groups.values());
  }

  static calculateMean(values) {
    return values.reduce((a, b) => a + b, 0) / values.length;
  }

  static calculateStdDev(values, mean) {
    const squareDiffs = values.map(value => Math.pow(value - mean, 2));
    return Math.sqrt(squareDiffs.reduce((a, b) => a + b, 0) / values.length);
  }
}

export default AnomalyDetectionService;

// src/components/stateAdmin/FraudNetworkVisualization.jsx
import React, { useEffect, useRef } from 'react';
import * as d3 from 'd3';

function FraudNetworkVisualization({ network, onNodeClick }) {
  const svgRef = useRef(null);

  useEffect(() => {
    if (!network || !svgRef.current) return;

    const svg = d3.select(svgRef.current);
    svg.selectAll('*').remove();

    const width = 500;
    const height = 300;

    svg.attr('width', width).attr('height', height);

    const simulation = d3.forceSimulation(network.nodes)
      .force('link', d3.forceLink(network.edges)
        .id(d => d.id)
        .distance(60))
      .force('charge', d3.forceManyBody().strength(-100))
      .force('center', d3.forceCenter(width / 2, height / 2));

    // Draw edges
    const edgeColors = {
      same_gps: '#f44336',
      same_photo: '#ff9800',
      time_correlation: '#2196f3'
    };

    const links = svg.append('g')
      .selectAll('line')
      .data(network.edges)
      .enter()
      .append('line')
      .attr('stroke', d => edgeColors[d.type] || '#999')
      .attr('stroke-width', d => Math.max(1, d.weight * 3))
      .attr('stroke-dasharray', d => d.type === 'time_correlation' ? '5,5' : null);

    // Draw nodes
    const nodes = svg.append('g')
      .selectAll('circle')
      .data(network.nodes)
      .enter()
      .append('circle')
      .attr('r', d => 8 + Math.sqrt(d.submissions) * 2)
      .attr('fill', d => d.confirmed ? '#f44336' : '#2196f3')
      .attr('stroke', '#fff')
      .attr('stroke-width', 2)
      .style('cursor', 'pointer')
      .on('click', (event, d) => onNodeClick && onNodeClick(d))
      .call(d3.drag()
        .on('start', dragstarted)
        .on('drag', dragged)
        .on('end', dragended));

    // Add labels
    const labels = svg.append('g')
      .selectAll('text')
      .data(network.nodes)
      .enter()
      .append('text')
      .text(d => d.name.split(' ')[0])
      .attr('font-size', 10)
      .attr('dx', 12)
      .attr('dy', 4);

    simulation.on('tick', () => {
      links
        .attr('x1', d => d.source.x)
        .attr('y1', d => d.source.y)
        .attr('x2', d => d.target.x)
        .attr('y2', d => d.target.y);

      nodes
        .attr('cx', d => d.x)
        .attr('cy', d => d.y);

      labels
        .attr('x', d => d.x)
        .attr('y', d => d.y);
    });

    function dragstarted(event) {
      if (!event.active) simulation.alphaTarget(0.3).restart();
      event.subject.fx = event.subject.x;
      event.subject.fy = event.subject.y;
    }

    function dragged(event) {
      event.subject.fx = event.x;
      event.subject.fy = event.y;
    }

    function dragended(event) {
      if (!event.active) simulation.alphaTarget(0);
      event.subject.fx = null;
      event.subject.fy = null;
    }

    return () => simulation.stop();
  }, [network]);

  return (
    <div className="fraud-network-visualization">
      <svg ref={svgRef} />
      <div className="network-legend">
        <span className="legend-item">
          <span className="line solid red" /> Same GPS
        </span>
        <span className="legend-item">
          <span className="line solid orange" /> Same Photo
        </span>
        <span className="legend-item">
          <span className="line dashed blue" /> Time Correlation
        </span>
      </div>
    </div>
  );
}

export default FraudNetworkVisualization;
